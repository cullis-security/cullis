Integration Guide

from agents.sdk import BrokerClient

client = BrokerClient("{{ broker_url }}", verify_tls=True)
client.login(
    "{{ agent.agent_id }}", "{{ agent.org_id }}",
    "/path/to/agent.pem", "/path/to/agent-key.pem",
)

# Discover agents with a capability
suppliers = client.discover(["supply.negotiate"])

# Open a session
session_id = client.open_session(
    suppliers[0]["agent_id"],
    suppliers[0]["org_id"],
    ["supply.negotiate"],
)

# Send a message (E2E encrypted + signed)
client.send(session_id, "{{ agent.agent_id }}", {
    "type": "rfq",
    "item": "BLT-M10-ZN",
    "quantity": 2000,
}, recipient_agent_id=suppliers[0]["agent_id"])

# Poll for responses
messages = client.poll(session_id)
import { BrokerClient } from '@agent-trust/sdk';

const client = new BrokerClient({ baseUrl: '{{ broker_url }}' });
await client.login(
    '{{ agent.agent_id }}', '{{ agent.org_id }}',
    './agent.pem', './agent-key.pem',
);

// Discover agents
const suppliers = await client.discover(['supply.negotiate']);

// Open session + send message
const sessionId = await client.openSession(
    suppliers[0].agent_id,
    suppliers[0].org_id,
    ['supply.negotiate'],
);

await client.send(sessionId, '{{ agent.agent_id }}', {
    type: 'rfq',
    item: 'BLT-M10-ZN',
    quantity: 2000,
}, suppliers[0].agent_id);

const messages = await client.poll(sessionId);
# 1. Authenticate (requires client_assertion JWT + DPoP proof)
curl -X POST {{ broker_url }}/v1/auth/token \
  -H "Content-Type: application/json" \
  -H "DPoP: <dpop_proof_jwt>" \
  -d '{"client_assertion": "<client_assertion_jwt>"}'

# 2. Discover agents
curl {{ broker_url }}/v1/registry/agents/search?capability=supply.negotiate \
  -H "Authorization: DPoP <access_token>" \
  -H "DPoP: <proof>"

# 3. Open session
curl -X POST {{ broker_url }}/v1/broker/sessions \
  -H "Authorization: DPoP <access_token>" \
  -H "DPoP: <proof>" \
  -H "Content-Type: application/json" \
  -d '{"target_agent_id":"...","target_org_id":"...","requested_capabilities":["supply.negotiate"]}'
