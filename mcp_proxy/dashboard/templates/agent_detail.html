import httpx

PROXY_URL = "{{ proxy_url }}"
# ADR-014: present the agent's TLS client cert at the handshake — that
# IS the credential. Mount cert.pem + key.pem from the identity bundle.
CERT = ("/path/to/agent.pem", "/path/to/agent-key.pem")
client = httpx.Client(cert=CERT, verify="/path/to/org-ca.crt")

headers = {"Content-Type": "application/json"}

# 1. Discover remote agents
resp = client.post(f"{PROXY_URL}/v1/egress/discover",
    headers=headers,
    json={"capabilities": ["order.read"]})
agents = resp.json()["agents"]
print(f"Found {len(agents)} agent(s)")

# 2. Open session with a remote agent
resp = client.post(f"{PROXY_URL}/v1/egress/sessions",
    headers=headers,
    json={
        "target_agent_id": agents[0]["agent_id"],
        "target_org_id": agents[0]["org_id"],
        "capabilities": ["order.read"]
    })
session_id = resp.json()["session_id"]

# 3. Send a message
client.post(f"{PROXY_URL}/v1/egress/send",
    headers=headers,
    json={
        "session_id": session_id,
        "payload": {"text": "Hello from {{ agent.agent_id }}"},
        "recipient_agent_id": agents[0]["agent_id"]
    })

# 4. Poll for responses
resp = client.get(f"{PROXY_URL}/v1/egress/messages/{session_id}",
    headers=headers)
for msg in resp.json()["messages"]:
    print(msg["payload"])

import https from "node:https";
import fs from "node:fs";

const PROXY_URL = "{{ proxy_url }}";
const agent = new https.Agent({
  cert: fs.readFileSync("/path/to/agent.pem"),
  key:  fs.readFileSync("/path/to/agent-key.pem"),
  ca:   fs.readFileSync("/path/to/org-ca.crt"),
});
const headers = {"Content-Type": "application/json"};

const fetchOpts = (method, body) => ({method, headers, agent, body});

// Discover agents
const agents = await fetch(`${PROXY_URL}/v1/egress/discover`,
  fetchOpts("POST", JSON.stringify({capabilities: ["order.read"]}))
).then(r => r.json()).then(d => d.agents);

// Open session
const {session_id} = await fetch(`${PROXY_URL}/v1/egress/sessions`,
  fetchOpts("POST", JSON.stringify({
    target_agent_id: agents[0].agent_id,
    target_org_id: agents[0].org_id,
    capabilities: ["order.read"]
  }))
).then(r => r.json());

// Send message
await fetch(`${PROXY_URL}/v1/egress/send`,
  fetchOpts("POST", JSON.stringify({
    session_id, payload: {text: "Hello from {{ agent.agent_id }}"},
    recipient_agent_id: agents[0].agent_id
  }))
);

// Poll messages
const {messages} = await fetch(
  `${PROXY_URL}/v1/egress/messages/${session_id}`,
  {headers, agent}
).then(r => r.json());
